#include<stdio.h>

#define SIZE 28
class Morse{
  public:
    Morse(int pin);
    int dot();
    int dash();
    int c_space();
    int w_space();
  private:
    int _pin;
    int _dottime;
};
Morse::Morse(int pin)
{
	pinMode(pin,OUTPUT);
	_pin=pin;
	_dottime=250;
}int Morse::dot()
{
	digitalWrite(_pin,HIGH);
	delay(_dottime);
	digitalWrite(_pin,LOW);
	delay(_dottime);
}
int Morse::dash(){
	digitalWrite(_pin,HIGH);
	delay(_dottime*4);
	digitalWrite(_pin,LOW);
	delay(_dottime);
}
int Morse::c_space()
{
	digitalWrite(_pin,LOW);
	delay(_dottime*3);
}
int Morse::w_space(){
	digitalWrite(_pin,LOW);
	delay(_dottime*7);
}
